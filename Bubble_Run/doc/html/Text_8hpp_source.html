157 void RenderText(GLuint &id, std::map<char, Text> &Characters, std::string text, float x, float y, float scale, glm::vec3 color)
158 {
159 // activate corresponding render state
160 glUniform3f(glGetUniformLocation(id, "textColor"), color.x, color.y, color.z);
161 glActiveTexture(GL_TEXTURE0);
162 glBindVertexArray(m_vao);
163
164 // iterate through all characters
165 std::string::const_iterator c;
166 for (c = text.begin(); c != text.end(); c++)
167 {
168 Text ch = Characters[*c];
169
170 float xpos = x + ch.getBearing().x * scale;
171 float ypos = y - (ch.getSize().y - ch.getBearing().y) * scale;
172
173 float w = ch.getSize().x * scale;
174 float h = ch.getSize().y * scale;
175 // update m_vbo for each character
176 float vertices[6][4] = {
177 {xpos, ypos + h, 0.0f, 0.0f},
178 {xpos, ypos, 0.0f, 1.0f},
179 {xpos + w, ypos, 1.0f, 1.0f},
180
181 {xpos, ypos + h, 0.0f, 0.0f},
182 {xpos + w, ypos, 1.0f, 1.0f},
183 {xpos + w, ypos + h, 1.0f, 0.0f}};
184 // render glyph texture over quad
185 glBindTexture(GL_TEXTURE_2D, ch.getTextureID());
186 // update content of m_vbo memory
187 glBindBuffer(GL_ARRAY_BUFFER, m_vbo);
188 glBufferSubData(GL_ARRAY_BUFFER, 0, sizeof(vertices), vertices);
189 glBindBuffer(GL_ARRAY_BUFFER, 0);
190 // render quad
191 glDrawArrays(GL_TRIANGLES, 0, 6);
192 // now advance cursors for next glyph (note that advance is number of 1/64 pixels)
193 x += (ch.getAdvance() >> 6) * scale; // bitshift by 6 to get value in pixels (2^6 = 64)
194 }
195 glBindVertexArray(0);
196 glBindTexture(GL_TEXTURE_2D, 0);
197 }
